location payg_cas03
	run-once 1
	priority 1
	call {
		release rest

	}
}


automacro rest {
location pay_gld 
	run-once 1
	priority 1
	call {
		do pm fusion_lumina $.hour:$.minute died at urpay
		release aturpayemp
		release aturpay
	}
}

automacro checkPlayer {
	location prtg_cas03
	timeout 1
	call {
		if (@eval(@playersID) > 0) {
			lock playAlive
			release playDead
			release autoScan
		} else {
			lock playDead
			release playAlive
		}
	}
}

automacro autoScan {
	disabled 1
	run-once 1
	timeout 15
	call scan
}

macro scan {
	#$x = @eval("a".","."b")
	#$list = @eval(@{$::playersList->getItems()})
	#log $list $x @player(fusion_lumina) @eval($::players{$::playersID[0]})
	lock checkPlayer
	$i = 0
	$j = @eval(@playersID)
	while ($i < $j) as loop
		do pm fusion_lumina urpay @eval($::players{$::playersID[$i]}) @eval($::players{$::playersID[$i]}->{guild}{name})
		$i++
	end loop
	release checkPlayer
}

automacro playAlive {
	disabled 1
	run-once 1
	status Play Dead
	call {
		do ss 143
	}
}

automacro playDead {
	disabled 1
	run-once 1
	status not Play Dead
	location not pay_gld
	call {
		do ss 143
	}
}
