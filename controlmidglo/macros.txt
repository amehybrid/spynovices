automacro checkPlayer {
	location te_prtcas03
	timeout 1
	call {
		if (@eval(@playersID) > 0) {
			lock playAlive
			release playDead
			release autoScan
		} else {
			lock playDead
			release playAlive
		}
	}
}

automacro autoScan {
	disabled 1
	run-once 1
	timeout 15
	call scan
}

macro scan {
	#$x = @eval("a".","."b")
	#$list = @eval(@{$::playersList->getItems()})
	#log $list $x @player(fusion_lumina) @eval($::players{$::playersID[0]})
	lock checkPlayer
	$i = 0
	$j = @eval(@playersID)
	while ($i < $j) as loop
		do pm fusion_lumina midglo @eval($::players{$::playersID[$i]}) @eval($::players{$::playersID[$i]}->{guild}{name})
		$i++
	end loop
	release checkPlayer
}

automacro playAlive {
	disabled 1
	run-once 1
	status Play Dead
	call {
		do ss 143
	}
}

automacro playDead {
	disabled 1
	run-once 1
	status not Play Dead
	location not prt_gld
	location not te_prt_gld
	call {
		do ss 143
	}
}
